** Gamma function

#+BEGIN_SRC clojure :noweb-ref impl
  (defn p-sum
    [idx n z]
    (transduce
     (map #(/ (idx %) (+ z %)))
     + (first idx) (range 1 n)))

  (defn log
    [z]
    (if (>= z 0.0)
      (let [x  (p-sum PLN (count PLN) z)
            z' (+ z 0.5)
            t  (+ z' GLN)]
        (- (+ (- (+ (* 0.5 (Math/log TWO_PI))
                    (* z' (Math/log t)))
                 t)
              (Math/log x))
           (Math/log z)))
      0.0))

  (defn gamma
    [z]
    (cond
      (< z 0.5) (* (/ PI (Math/sin (* z PI))) (gamma (- 1 z)))
      (> z 100) (Math/exp (log z))
      :else     (let [z  (dec z)
                      z' (+ z 0.5)
                      t  (+ z' 7)]
                  (* (* (* (Math/sqrt TWO_PI)
                           (Math/pow t z'))
                        (Math/exp (- t)))
                     (p-sum P 9 z)))))
